pub const BATCH_MATRIX_VECTOR_MULTIPLY_SHADER: &str = "// WebGPU Compute Shader: Batch Matrix-Vector Multiplication\n// Efficiently processes multiple vectors against the same matrix\n// Optimized for neural network batch inference\n\n@group(0) @binding(0) var<storage, read> matrix: array<f32>;\n@group(0) @binding(1) var<storage, read> vectors: array<f32>;\n@group(0) @binding(2) var<storage, read_write> results: array<f32>;\n\nstruct Uniforms {\n    rows: u32,\n    cols: u32,\n    batch_size: u32,\n    reserved: u32,\n}\n\n@group(0) @binding(3) var<uniform> uniforms: Uniforms;\n\n// 2D workgroup layout: 16x16 = 256 threads per workgroup\n// X dimension: matrix rows, Y dimension: batch items\n@compute @workgroup_size(16, 16)\nfn main(@builtin(global_invocation_id) global_id: vec3<u32>) {\n    let row = global_id.x;\n    let batch_idx = global_id.y;\n    \n    // Bounds check\n    if (row >= uniforms.rows || batch_idx >= uniforms.batch_size) {\n        return;\n    }\n    \n    // Compute dot product for this row and batch item\n    var sum: f32 = 0.0;\n    \n    let cols = uniforms.cols;\n    let matrix_row_offset = row * cols;\n    let vector_offset = batch_idx * cols;\n    \n    // Vectorized processing for better memory bandwidth utilization\n    let vectorized_cols = cols & ~3u; // Round down to multiple of 4\n    \n    for (var col = 0u; col < vectorized_cols; col += 4u) {\n        let matrix_base = matrix_row_offset + col;\n        let vector_base = vector_offset + col;\n        \n        // Load 4 matrix elements (same for all batches)\n        let m0 = matrix[matrix_base];\n        let m1 = matrix[matrix_base + 1u];\n        let m2 = matrix[matrix_base + 2u];\n        let m3 = matrix[matrix_base + 3u];\n        \n        // Load 4 vector elements (batch-specific)\n        let v0 = vectors[vector_base];\n        let v1 = vectors[vector_base + 1u];\n        let v2 = vectors[vector_base + 2u];\n        let v3 = vectors[vector_base + 3u];\n        \n        // Fused multiply-add\n        sum += m0 * v0 + m1 * v1 + m2 * v2 + m3 * v3;\n    }\n    \n    // Handle remaining elements\n    for (var col = vectorized_cols; col < cols; col++) {\n        sum += matrix[matrix_row_offset + col] * vectors[vector_offset + col];\n    }\n    \n    // Store result: results[batch_idx * rows + row]\n    let result_index = batch_idx * uniforms.rows + row;\n    results[result_index] = sum;\n}";
