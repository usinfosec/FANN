23/// GPU device capabilities and optimization parameters
24#[derive(Debug, Clone)]
25pub struct GpuCapabilities {
26    /// Maximum workgroup size in each dimension
27    pub max_workgroup_size: [u32; 3],
28
29    /// Maximum number of threads per workgroup
30    pub max_threads_per_workgroup: u32,
31
32    /// Preferred workgroup size for this GPU architecture
33    pub preferred_workgroup_size: u32,
34
35    /// Memory bandwidth in GB/s
36    pub memory_bandwidth_gbps: f32,
37
38    /// Number of compute units/streaming multiprocessors
39    pub compute_units: u32,
40
41    /// Maximum shared memory per workgroup in bytes
42    pub max_shared_memory_bytes: u32,
43
44    /// Whether the GPU supports subgroups/wavefronts
45    pub supports_subgroups: bool,
46
47    /// Subgroup/wavefront size (typically 32 or 64)
48    pub subgroup_size: u32,
49}

51/// Optimized kernel configuration for specific operations
52#[derive(Debug, Clone)]
53pub struct KernelConfig {

56
57    /// Number of workgroups to dispatch
58    pub num_workgroups: [u32; 3],
59
60    /// Whether to use vectorized memory access
61    pub use_vectorized_access: bool,
62
63    /// Tile size for tiled algorithms
64    pub tile_size: u32,
65
66    /// Number of elements processed per thread
67    pub elements_per_thread: u32,
68
69    /// Estimated performance in GFLOPS
70    pub estimated_gflops: f32,
71}

73/// Performance optimization metrics
74#[derive(Debug, Clone)]
75pub struct OptimizationMetrics {
76    /// Memory bandwidth utilization (0.0 to 1.0)
77    pub memory_utilization: f32,
78
79    /// Compute utilization (0.0 to 1.0)
80    pub compute_utilization: f32,

82    /// Occupancy (threads per SM / max threads per SM)

131    /// Optimize kernel configuration for matrix-vector multiplication
132    pub fn optimize_matrix_vector_multiply(
133        &mut self,
134        rows: usize,
135        cols: usize,
136    ) -> Result<KernelConfig, ComputeError> {
137        let cache_key = (ShaderType::MatrixVectorMultiply, rows * cols);
138
139        // Check cache first
140        if let Some(cached_config) = self.config_cache.get(&cache_key) {
141            return Ok(cached_config.clone());
142        }
143
144        // Calculate optimal configuration
145        let config = self.calculate_matrix_vector_config(rows, cols)?;
146
147        // Cache the result
148        self.config_cache.insert(cache_key, config.clone());
149
150        Ok(config)
151    }

193    /// Record performance metrics for continuous optimization
194    pub fn record_performance(
195        &mut self,
196        shader_type: ShaderType,
197        data_size: usize,
198        metrics: OptimizationMetrics,
199    ) {
200        let key = (shader_type, data_size);
201        let history = self.performance_history.entry(key.clone()).or_default();
202        history.push(metrics);
203
204        // Keep only recent history (last 10 measurements)
205        if let Some(history) = self.performance_history.get_mut(&key) {
206            if history.len() > 10 {
207                history.remove(0);
208            }
209        }
210    }

212    /// Get performance predictions based on historical data
213    pub fn predict_performance(
214        &self,
215        shader_type: &ShaderType,
216        data_size: usize,
217    ) -> Option<OptimizationMetrics> {
218        let key = (shader_type.clone(), data_size);
219
220        if let Some(history) = self.performance_history.get(&key) {
221            if !history.is_empty() {
222                // Return average of recent measurements
223                let count = history.len() as f32;
224                let avg_memory_util =
225                    history.iter().map(|m| m.memory_utilization).sum::<f32>() / count;
226                let avg_compute_util =
227                    history.iter().map(|m| m.compute_utilization).sum::<f32>() / count;
228                let avg_occupancy = history.iter().map(|m| m.occupancy).sum::<f32>() / count;
229                let avg_memory_eff =
230                    history.iter().map(|m| m.memory_efficiency).sum::<f32>() / count;
231                let avg_exec_time = history
232                    .iter()
233                    .map(|m| m.estimated_execution_time_us)
234                    .sum::<f32>()
235                    / count;
236
237                return Some(OptimizationMetrics {
238                    memory_utilization: avg_memory_util,
239                    compute_utilization: avg_compute_util,
240                    occupancy: avg_occupancy,
241                    memory_efficiency: avg_memory_eff,
242                    estimated_execution_time_us: avg_exec_time,
243                });
244            }
245        }
246
247        None
248    }

250    /// Auto-tune workgroup size by testing different configurations
251    pub fn auto_tune_workgroup_size(
252        &mut self,
253        shader_type: ShaderType,
254        data_size: usize,
255    ) -> Result<[u32; 3], ComputeError> {
256        let test_sizes = vec![64, 128, 256, 512, 1024];
257        let mut best_config = [256, 1, 1]; // Default
258        let mut best_score = 0.0f32;
259
260        for size in test_sizes {
261            if size <= self.gpu_capabilities.max_threads_per_workgroup {
262                let config = [size, 1, 1];
263                let score = self.evaluate_workgroup_config(&shader_type, data_size, config);
264
265                if score > best_score {
266                    best_score = score;
267                    best_config = config;
268                }
269            }
270        }
271
272        Ok(best_config)
273    }

288    fn calculate_matrix_vector_config(
289        &self,
290        rows: usize,
291        _cols: usize,
292    ) -> Result<KernelConfig, ComputeError> {
293        // Optimize for memory bandwidth and occupancy
294        let workgroup_size = if rows >= 1024 {
295            [256, 1, 1] // Large problems - maximize parallelism
296        } else if rows >= 256 {
297            [128, 1, 1] // Medium problems - balance parallelism and overhead
298        } else {
299            [64, 1, 1] // Small problems - avoid underutilization
300        };
301
302        let num_workgroups = [
303            ((rows as u32 + workgroup_size[0] - 1) / workgroup_size[0]),
304            1,
305            1,
306        ];
307
308        Ok(KernelConfig {
309            workgroup_size,
310            num_workgroups,
311            use_vectorized_access: true,
312            tile_size: 16,
313            elements_per_thread: 4,
314            estimated_gflops: self.estimate_matrix_vector_gflops(rows),
315        })
316    }
317
318    fn calculate_batch_matrix_vector_config(
319        &self,
320        rows: usize,
321        _cols: usize,
322        batch_size: usize,
323    ) -> Result<KernelConfig, ComputeError> {
324        // 2D workgroup layout for batch processing
325        let total_work = rows * batch_size;
326
327        let workgroup_size = if total_work >= 4096 {
328            [16, 16, 1] // 256 threads total, good for large batches
329        } else if total_work >= 1024 {
330            [16, 8, 1] // 128 threads total
331        } else {
332            [8, 8, 1] // 64 threads total for small batches
333        };
334
335        let num_workgroups = [
336            ((rows as u32 + workgroup_size[0] - 1) / workgroup_size[0]),
337            ((batch_size as u32 + workgroup_size[1] - 1) / workgroup_size[1]),
338            1,
339        ];
340
341        Ok(KernelConfig {
342            workgroup_size,
343            num_workgroups,
344            use_vectorized_access: true,
345            tile_size: 16,
346            elements_per_thread: 1,
347            estimated_gflops: self.estimate_batch_matrix_vector_gflops(rows, batch_size),
348        })
349    }

351    fn calculate_activation_config(
352        &self,
353        _shader_type: ShaderType,
354        vector_size: usize,
355    ) -> Result<KernelConfig, ComputeError> {
356        // Activation functions are memory-bound, optimize for memory bandwidth
357        let workgroup_size = if vector_size >= 2048 {
358            [256, 1, 1] // Maximize memory bandwidth utilization
359        } else if vector_size >= 512 {
360            [128, 1, 1]
361        } else {
362            [64, 1, 1]
363        };
364
365        let num_workgroups = [
366            ((vector_size as u32 + workgroup_size[0] - 1) / workgroup_size[0]),
367            1,
368            1,
369        ];
370
371        Ok(KernelConfig {
372            workgroup_size,
373            num_workgroups,
374            use_vectorized_access: true,
375            tile_size: 1, // Not applicable for element-wise operations
376            elements_per_thread: 1,
377            estimated_gflops: self.estimate_activation_gflops(vector_size),
378        })
379    }
380
381    fn evaluate_workgroup_config(
382        &self,
383        _shader_type: &ShaderType,
384        data_size: usize,
385        workgroup_config: [u32; 3],
386    ) -> f32 {
387        let threads_per_workgroup = workgroup_config[0] * workgroup_config[1] * workgroup_config[2];
388        let num_workgroups = (data_size as u32 + threads_per_workgroup - 1) / threads_per_workgroup;
389
390        // Calculate occupancy
391        let max_workgroups_per_sm =
392            self.gpu_capabilities.max_threads_per_workgroup / threads_per_workgroup;
393        let occupancy = (max_workgroups_per_sm.min(8) as f32) / 8.0; // Assume 8 max workgroups per SM
394
395        // Simple scoring function (higher is better)
396        let memory_efficiency = if threads_per_workgroup >= 32 {
397            1.0
398        } else {
399            threads_per_workgroup as f32 / 32.0
400        };
401        let parallelism_score = (num_workgroups as f32)
402            .min(self.gpu_capabilities.compute_units as f32)
403            / self.gpu_capabilities.compute_units as f32;
404
405        occupancy * 0.4 + memory_efficiency * 0.3 + parallelism_score * 0.3
406    }

408    fn estimate_matrix_vector_gflops(&self, rows: usize) -> f32 {
409        // Rough estimate based on GPU capabilities
410        let base_gflops = self.gpu_capabilities.compute_units as f32 * 100.0; // 100 GFLOPS per CU
411        let problem_efficiency = if rows >= 1024 {
412            0.8
413        } else {
414            rows as f32 / 1024.0 * 0.8
415        };
416        base_gflops * problem_efficiency
417    }
418
419    fn estimate_batch_matrix_vector_gflops(&self, rows: usize, batch_size: usize) -> f32 {
420        let single_gflops = self.estimate_matrix_vector_gflops(rows);
421        let batch_efficiency = (batch_size as f32).min(16.0) / 16.0; // Diminishing returns after 16
422        single_gflops * batch_size as f32 * batch_efficiency
423    }
424
425    fn estimate_activation_gflops(&self, vector_size: usize) -> f32 {
426        // Activation functions are simpler, memory-bound operations
427        let memory_bound_gflops = self.gpu_capabilities.memory_bandwidth_gbps * 4.0; // 4 GFLOPS per GB/s
428        let problem_efficiency = (vector_size as f32 / 1024.0).min(1.0);
429        memory_bound_gflops * problem_efficiency
430    }
