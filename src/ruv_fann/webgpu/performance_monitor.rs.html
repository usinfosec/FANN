195    /// Record a performance measurement
196    pub fn record_measurement(
197        &mut self,
198        measurement: PerformanceMeasurement,
199    ) -> Vec<PerformanceAlert> {
200        let key = (measurement.shader_type.clone(), measurement.data_size);
201        let mut alerts = Vec::new();
202
203        // Add measurement to history
204        let measurements = self.measurements.entry(key.clone()).or_default();
205        measurements.push(measurement.clone());
206
207        // Limit history size
208        if measurements.len() > self.max_measurements_per_operation {
209            measurements.remove(0);
210        }
211
212        // Invalidate stats cache for this operation
213        self.stats_cache.remove(&key);
214
215        // Check for performance alerts
216        let measurements_len = measurements.len();
217        if measurements_len >= self.min_samples_for_stats {
218            alerts.extend(self.check_performance_alerts(&key, &measurement));
219        }
220
221        // Update real-time metrics
222        self.update_real_time_metrics(&measurement);
223
224        alerts
225    }

227    /// Get performance statistics for an operation
228    pub fn get_stats(
229        &mut self,
230        shader_type: &ShaderType,
231        data_size: usize,
232    ) -> Option<PerformanceStats> {
233        let key = (shader_type.clone(), data_size);
234
235        // Check cache first
236        if let Some(cached_stats) = self.stats_cache.get(&key) {
237            return Some(cached_stats.clone());
238        }
239
240        // Calculate stats if we have enough measurements
241        if let Some(measurements) = self.measurements.get(&key) {
242            if measurements.len() >= self.min_samples_for_stats {
243                let stats = self.calculate_stats(measurements);
244                self.stats_cache.insert(key, stats.clone());
245                return Some(stats);
246            }
247        }
248
249        None
250    }

257    /// Get performance prediction for an operation
258    pub fn predict_performance(
259        &self,
260        shader_type: &ShaderType,
261        data_size: usize,
262    ) -> Option<OptimizationMetrics> {
263        let key = (shader_type.clone(), data_size);
264
265        if let Some(measurements) = self.measurements.get(&key) {
266            if !measurements.is_empty() {
267                let latest = &measurements[measurements.len() - 1];
268
269                return Some(OptimizationMetrics {
270                    memory_utilization: latest.memory_efficiency,
271                    compute_utilization: latest.gpu_occupancy,
272                    occupancy: latest.gpu_occupancy,
273                    memory_efficiency: latest.memory_efficiency,
274                    estimated_execution_time_us: latest.execution_time.as_micros() as f32,
275                });
276            }
277        }
278
279        None
280    }

282    /// Get performance trend for an operation
283    pub fn get_performance_trend(
284        &self,
285        shader_type: &ShaderType,
286        data_size: usize,
287    ) -> PerformanceTrend {
288        let key = (shader_type.clone(), data_size);
289
290        if let Some(measurements) = self.measurements.get(&key) {
291            if measurements.len() < 3 {
292                return PerformanceTrend::Insufficient;
293            }
294
295            // Compare recent performance to older performance
296            let recent_count = measurements.len().min(5);
297            let older_count = measurements.len().min(10) - recent_count;
298
299            if older_count == 0 {
300                return PerformanceTrend::Insufficient;
301            }
302
303            let recent_avg = measurements
304                .iter()
305                .rev()
306                .take(recent_count)
307                .map(|m| m.execution_time.as_nanos() as f64)
308                .sum::<f64>()
309                / recent_count as f64;
310
311            let older_avg = measurements
312                .iter()
313                .rev()
314                .skip(recent_count)
315                .take(older_count)
316                .map(|m| m.execution_time.as_nanos() as f64)
317                .sum::<f64>()
318                / older_count as f64;
319
320            let improvement_ratio = older_avg / recent_avg;
321
322            if improvement_ratio > 1.05 {
323                PerformanceTrend::Improving
324            } else if improvement_ratio < 0.95 {
325                PerformanceTrend::Degrading
326            } else {
327                PerformanceTrend::Stable
328            }
329        } else {
330            PerformanceTrend::Insufficient
331        }
332    }

348    fn update_real_time_metrics(&mut self, measurement: &PerformanceMeasurement) {
349        self.real_time_metrics.total_operations += 1;
350        self.real_time_metrics.total_gpu_time += measurement.execution_time;
351
352        // Update rolling averages (simplified)
353        let alpha = 0.1; // Smoothing factor
354        self.real_time_metrics.rolling_avg_memory_bandwidth = alpha
355            * measurement.memory_bandwidth_gbps
356            + (1.0 - alpha) * self.real_time_metrics.rolling_avg_memory_bandwidth;
357
358        self.real_time_metrics.rolling_avg_compute_throughput = alpha
359            * measurement.compute_throughput_gflops
360            + (1.0 - alpha) * self.real_time_metrics.rolling_avg_compute_throughput;
361
362        // Calculate operations per second (over last second)
363        let total_seconds = self.real_time_metrics.total_gpu_time.as_secs_f32();
364        if total_seconds > 0.0 {
365            self.real_time_metrics.current_ops_per_second =
366                self.real_time_metrics.total_operations as f32 / total_seconds;
367        }
368
369        // GPU utilization approximation
370        self.real_time_metrics.gpu_utilization_percent =
371            (measurement.gpu_occupancy * 100.0).min(100.0);
372    }

374    fn calculate_stats(&self, measurements: &[PerformanceMeasurement]) -> PerformanceStats {
375        let count = measurements.len();
376
377        // Calculate execution time statistics
378        let execution_times: Vec<u64> = measurements
379            .iter()
380            .map(|m| m.execution_time.as_nanos() as u64)
381            .collect();
382
383        let avg_execution_nanos = execution_times.iter().sum::<u64>() / count as u64;
384        let min_execution_nanos = *execution_times.iter().min().unwrap();
385        let max_execution_nanos = *execution_times.iter().max().unwrap();
386
387        // Calculate standard deviation
388        let variance = execution_times
389            .iter()
390            .map(|&time| {
391                let diff = time as f64 - avg_execution_nanos as f64;
392                diff * diff
393            })
394            .sum::<f64>()
395            / count as f64;
396        let stddev_nanos = variance.sqrt() as u64;
397
398        // Calculate averages for other metrics
399        let avg_memory_bandwidth = measurements
400            .iter()
401            .map(|m| m.memory_bandwidth_gbps)
402            .sum::<f32>()
403            / count as f32;
404
405        let avg_compute_throughput = measurements
406            .iter()
407            .map(|m| m.compute_throughput_gflops)
408            .sum::<f32>()
409            / count as f32;
410
411        let avg_gpu_occupancy =
412            measurements.iter().map(|m| m.gpu_occupancy).sum::<f32>() / count as f32;
413
414        PerformanceStats {
415            sample_count: count,
416            avg_execution_time: Duration::from_nanos(avg_execution_nanos),
417            min_execution_time: Duration::from_nanos(min_execution_nanos),
418            max_execution_time: Duration::from_nanos(max_execution_nanos),
419            execution_time_stddev: Duration::from_nanos(stddev_nanos),
420            avg_memory_bandwidth_gbps: avg_memory_bandwidth,
421            avg_compute_throughput_gflops: avg_compute_throughput,
422            avg_gpu_occupancy,
423            trend: PerformanceTrend::Insufficient, // Will be calculated separately
424        }
425    }

427    fn check_performance_alerts(
428        &self,
429        key: &(ShaderType, usize),
430        measurement: &PerformanceMeasurement,
431    ) -> Vec<PerformanceAlert> {
432        let mut alerts = Vec::new();
433
434        if let Some(measurements) = self.measurements.get(key) {
435            if measurements.len() >= 2 {
436                let previous = &measurements[measurements.len() - 2];
437
438                // Check for execution time degradation
439                let time_ratio = measurement.execution_time.as_nanos() as f32
440                    / previous.execution_time.as_nanos() as f32;
441                if time_ratio > self.alert_thresholds.max_execution_time_degradation {
442                    alerts.push(PerformanceAlert {
443                        alert_type: AlertType::PerformanceDegradation,
444                        severity: if time_ratio > 2.0 {
445                            AlertSeverity::Critical
446                        } else {
447                            AlertSeverity::Warning
448                        },
449                        shader_type: measurement.shader_type.clone(),
450                        data_size: measurement.data_size,
451                        message: format!(
452                            "Execution time increased by {:.1}%",
453                            (time_ratio - 1.0) * 100.0
454                        ),
455                        timestamp: measurement.timestamp,
456                    });
457                }
458
459                // Check memory bandwidth
460                if measurement.memory_bandwidth_gbps
461                    < self.alert_thresholds.min_memory_bandwidth_utilization
462                {
463                    alerts.push(PerformanceAlert {
464                        alert_type: AlertType::LowMemoryBandwidth,
465                        severity: AlertSeverity::Info,
466                        shader_type: measurement.shader_type.clone(),
467                        data_size: measurement.data_size,
468                        message: format!(
469                            "Low memory bandwidth: {:.1} GB/s",
470                            measurement.memory_bandwidth_gbps
471                        ),
472                        timestamp: measurement.timestamp,
473                    });
474                }
475
476                // Check GPU occupancy
477                if measurement.gpu_occupancy < self.alert_thresholds.min_gpu_occupancy {
478                    alerts.push(PerformanceAlert {
479                        alert_type: AlertType::LowGpuOccupancy,
480                        severity: AlertSeverity::Warning,
481                        shader_type: measurement.shader_type.clone(),
482                        data_size: measurement.data_size,
483                        message: format!(
484                            "Low GPU occupancy: {:.1}%",
485                            measurement.gpu_occupancy * 100.0
486                        ),
487                        timestamp: measurement.timestamp,
488                    });
489                }
490            }
491        }
492
493        alerts
494    }
