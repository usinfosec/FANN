40/// ComputeContext manages backend selection and operation dispatch for `Network<T>`
41///
42/// This bridge provides seamless integration between `Network<T>` and the advanced
43/// WebGPU backend while maintaining full compatibility with existing code.
44#[derive(Debug)]
45pub struct ComputeContext<T: Float + std::fmt::Debug + Send + Sync + 'static> {
46    /// Backend selector for intelligent backend switching
47    backend_selector: BackendSelector<T>,
48    /// Current backend type being used
49    current_backend: BackendType,
50    /// WebGPU backend instance (when available)
51    #[cfg(feature = "gpu")]
52    webgpu_backend: Option<Arc<WebGPUBackend<T>>>,
53    #[cfg(not(feature = "gpu"))]
54    webgpu_backend: Option<()>,
55    /// GPU acceleration enabled flag
56    gpu_enabled: bool,
57    /// Performance tracking and optimization
58    performance_tracker: Arc<std::sync::Mutex<PerformanceTracker>>,
59    /// Cache for converted weights to avoid repeated conversions
60    weight_cache: std::collections::HashMap<usize, (Vec<T>, MatrixDims)>,
61}

83    pub async fn new() -> ComputeResult<Self> {
84        let backend_selector = BackendSelector::new();
85
86        // Try to initialize WebGPU backend
87        #[cfg(feature = "gpu")]
88        let (webgpu_backend, gpu_enabled) = match WebGPUBackend::<T>::initialize().await {
89            Ok(backend) => (Some(Arc::new(backend)), true),
90            Err(_) => (None, false),
91        };
92
93        #[cfg(not(feature = "gpu"))]
94        let (webgpu_backend, gpu_enabled) = (None, false);
95
96        // Select initial backend based on availability
97        let current_backend = if gpu_enabled {
98            BackendType::WebGPU
99        } else {
100            BackendType::Simd
101        };
102
103        Ok(Self {
104            backend_selector,
105            current_backend,
106            webgpu_backend,
107            gpu_enabled,
108            performance_tracker: Arc::new(std::sync::Mutex::new(PerformanceTracker::new())),
109            weight_cache: HashMap::new(),
110        })
111    }

135    /// Select optimal backend for given problem size
136    pub fn select_backend(&mut self, problem_size: usize) -> BackendType {
137        let profile = crate::webgpu::backend::ComputeProfile {
138            matrix_size: match problem_size {
139                0..=10000 => crate::webgpu::backend::MatrixSize::Small,
140                10001..=1000000 => crate::webgpu::backend::MatrixSize::Medium,
141                _ => crate::webgpu::backend::MatrixSize::Large,
142            },
143            batch_size: 1,
144            operation_type: crate::webgpu::backend::OperationType::Inference,
145        };
146
147        let selected = self
148            .backend_selector
149            .select_backend(&profile)
150            .map(|backend| backend.backend_type())
151            .unwrap_or(BackendType::Cpu);
152
153        // Only use GPU if it's actually available
154        if selected == BackendType::WebGPU && !self.is_gpu_available() {
155            self.current_backend = BackendType::Simd;
156        } else {
157            self.current_backend = selected;
158        }
159
160        self.current_backend
161    }

163    /// Convert Network layer to matrix format with caching
164    fn get_layer_weights(
165        &mut self,
166        layer: &Layer<T>,
167        layer_id: usize,
168    ) -> ComputeResult<(Vec<T>, MatrixDims)> {
169        // Check cache first
170        if let Some(cached) = self.weight_cache.get(&layer_id) {
171            return Ok(cached.clone());
172        }
173
174        // Debug layer information
175        println!("Converting layer {layer_id} to matrix format");
176        println!("  Layer has {} neurons", layer.neurons.len());
177
178        // In FANN networks, bias neurons are included in the layer
179        // We need to find non-bias neurons for the output
180        let non_bias_neurons: Vec<&crate::Neuron<T>> =
181            layer.neurons.iter().filter(|n| !n.is_bias).collect();
182
183        println!("  Layer has {} non-bias neurons", non_bias_neurons.len());
184
185        // Convert layer connections to matrix format
186        // In a FANN network, the input size is the number of connections on each neuron
187        // (all neurons should have the same number of connections)
188        let input_size = if let Some(neuron) = non_bias_neurons.first() {
189            println!(
190                "  First neuron has {} connections",
191                neuron.connections.len()
192            );
193            neuron.connections.len()
194        } else {
195            println!("  No non-bias neurons found!");
196            return Err(ComputeError::InvalidDimensions(format!(
197                "Layer {layer_id} has no non-bias neurons"
198            )));
199        };
200
201        let output_size = non_bias_neurons.len();
202
203        println!("  Matrix dimensions: {output_size}x{input_size} (output_size x input_size)");
204
205        if input_size == 0 || output_size == 0 {
206            return Err(ComputeError::InvalidDimensions(format!(
207                "Invalid layer dimensions: {output_size}x{input_size}"
208            )));
209        }
210
211        let mut weights = Vec::with_capacity(output_size * input_size);
212
213        // Build weight matrix row by row (each row = one output neuron's weights)
214        for neuron in &non_bias_neurons {
215            // Ensure we have enough connections
216            if neuron.connections.len() != input_size {
217                return Err(ComputeError::InvalidDimensions(format!(
218                    "Neuron has {} connections, expected {}",
219                    neuron.connections.len(),
220                    input_size
221                )));
222            }
223
224            // Add weights for this neuron to the matrix
225            for i in 0..input_size {
226                weights.push(neuron.connections[i].weight);
227            }
228        }
229
230        if weights.len() != output_size * input_size {
231            return Err(ComputeError::InvalidDimensions(format!(
232                "Weight matrix size mismatch: got {}, expected {}",
233                weights.len(),
234                output_size * input_size
235            )));
236        }
237
238        let dims = MatrixDims {
239            rows: output_size,
240            cols: input_size,
241        };
242        let result = (weights, dims);
243
244        // Cache the result
245        self.weight_cache.insert(layer_id, result.clone());
246
247        Ok(result)
248    }

250    /// Execute forward pass for a layer with optimal backend selection
251    pub async fn compute_layer_forward(
252        &mut self,
253        layer: &Layer<T>,
254        layer_id: usize,
255        inputs: &[T],
256    ) -> ComputeResult<Vec<T>>
257    where
258        T: Clone + num_traits::ToPrimitive + 'static,
259    {
260        let start_time = std::time::Instant::now();
261
262        // Get layer weights
263        let (weights, dims) = self.get_layer_weights(layer, layer_id)?;
264
265        // Check if we need to append a bias input (value 1.0)
266        let mut input_with_bias = inputs.to_vec();
267        if dims.cols == inputs.len() + 1 {
268            // The extra column is likely for the bias input (common in FANN architecture)
269            println!("  Adding bias input to match expected dimensions");
270            input_with_bias.push(T::one()); // Add bias input with value 1.0
271        } else if inputs.len() != dims.cols {
272            return Err(ComputeError::InvalidDimensions(format!(
273                "Input size {} doesn't match expected {} and doesn't match bias pattern",
274                inputs.len(),
275                dims.cols
276            )));
277        }
278
279        // Select optimal backend for this problem size
280        let problem_size = dims.rows * dims.cols;
281        let backend_type = self.select_backend(problem_size);
282
283        // Execute computation based on selected backend
284        let result = match backend_type {
285            BackendType::WebGPU if self.is_gpu_available() => {
286                self.compute_layer_gpu(layer, &weights, &input_with_bias, dims)
287                    .await
288            }
289            BackendType::Simd => {
290                self.compute_layer_simd(layer, &weights, &input_with_bias, dims)
291                    .await
292            }
293            _ => {
294                self.compute_layer_cpu(layer, &weights, &input_with_bias, dims)
295                    .await
296            }
297        };
298
299        // Record performance metrics
300        let duration = start_time.elapsed().as_secs_f64();
301        if let Ok(mut tracker) = self.performance_tracker.lock() {
302            tracker.record_operation("layer_forward", duration, backend_type);
303        }
304
305        result
306    }

308    /// GPU-accelerated layer computation
309    async fn compute_layer_gpu(
310        &self,
311        layer: &Layer<T>,
312        weights: &[T],
313        inputs: &[T],
314        dims: MatrixDims,
315    ) -> ComputeResult<Vec<T>>
316    where
317        T: Clone + num_traits::ToPrimitive + 'static,
318    {
319        #[cfg(feature = "gpu")]
320        {
321            if let Some(ref gpu_backend) = self.webgpu_backend {
322                // Matrix-vector multiplication
323                let outputs =
324                    gpu_backend.matrix_vector_multiply(weights, inputs, dims.rows, dims.cols)?;
325
326                // Apply activation function
327                // Get activation function from first non-bias neuron
328                let activation_function = layer
329                    .neurons
330                    .iter()
331                    .find(|n| !n.is_bias)
332                    .map(|n| n.activation_function)
333                    .unwrap_or(ActivationFunction::Linear);
334                let steepness = T::one();
335                gpu_backend.apply_activation_function(&outputs, activation_function, steepness)
336            } else {
337                Err(ComputeError::GpuUnavailable)
338            }
339        }
340
341        #[cfg(not(feature = "gpu"))]
342        {
343            Err(ComputeError::GpuUnavailable)
344        }
345    }

347    /// SIMD-optimized layer computation
348    async fn compute_layer_simd(
349        &self,
350        layer: &Layer<T>,
351        weights: &[T],
352        inputs: &[T],
353        dims: MatrixDims,
354    ) -> ComputeResult<Vec<T>>
355    where
356        T: Clone + 'static,
357    {
358        // Use backend selector to get SIMD backend
359        let profile = crate::webgpu::backend::ComputeProfile {
360            matrix_size: crate::webgpu::backend::MatrixSize::Medium,
361            batch_size: 1,
362            operation_type: crate::webgpu::backend::OperationType::Inference,
363        };
364
365        if let Some(backend) = self.backend_selector.select_backend(&profile) {
366            let outputs = backend.matrix_vector_multiply(weights, inputs, dims.rows, dims.cols)?;
367            // Get activation function from first non-bias neuron
368            let activation_function = layer
369                .neurons
370                .iter()
371                .find(|n| !n.is_bias)
372                .map(|n| n.activation_function)
373                .unwrap_or(ActivationFunction::Linear);
374            let steepness = T::one();
375            backend.apply_activation_function(&outputs, activation_function, steepness)
376        } else {
377            self.compute_layer_cpu(layer, weights, inputs, dims).await
378        }
379    }

381    /// CPU fallback layer computation
382    async fn compute_layer_cpu(
383        &self,
384        layer: &Layer<T>,
385        weights: &[T],
386        inputs: &[T],
387        dims: MatrixDims,
388    ) -> ComputeResult<Vec<T>> {
389        let mut outputs = Vec::with_capacity(dims.rows);
390
391        // Manual matrix-vector multiplication
392        for row in 0..dims.rows {
393            let mut sum = T::zero();
394            for col in 0..dims.cols {
395                sum = sum + weights[row * dims.cols + col] * inputs[col];
396            }
397            outputs.push(sum);
398        }
399
400        // Apply activation function
401        // Get activation function from first non-bias neuron
402        let activation_function = layer
403            .neurons
404            .iter()
405            .find(|n| !n.is_bias)
406            .map(|n| n.activation_function)
407            .unwrap_or(ActivationFunction::Linear);
408        let result: Vec<T> = outputs
409            .into_iter()
410            .map(|x| apply_activation_cpu(x, activation_function, T::one()))
411            .collect();
412
413        Ok(result)
414    }

416    /// Execute complete network forward pass with optimal backend coordination
417    pub async fn compute_network_forward(
418        &mut self,
419        network: &Network<T>,
420        inputs: &[T],
421    ) -> ComputeResult<Vec<T>>
422    where
423        T: Clone + num_traits::ToPrimitive + 'static,
424    {
425        // Validate network has layers
426        if network.layers.is_empty() {
427            return Err(ComputeError::InvalidDimensions(
428                "Network has no layers".to_string(),
429            ));
430        }
431
432        // Validate input size matches input layer (excluding bias neuron)
433        if !network.layers.is_empty() && inputs.len() != network.num_inputs() {
434            return Err(ComputeError::InvalidDimensions(format!(
435                "Input size {} doesn't match network input size {}",
436                inputs.len(),
437                network.num_inputs()
438            )));
439        }
440
441        let mut current_inputs = inputs.to_vec();
442
443        // Process each layer, starting from the first hidden layer (index 1)
444        // The input layer (index 0) is just for passing inputs
445        for (layer_id, layer) in network.layers.iter().enumerate().skip(1) {
446            // Skip input layer (index 0)
447            current_inputs = match self
448                .compute_layer_forward(layer, layer_id, &current_inputs)
449                .await
450            {
451                Ok(outputs) => outputs,
452                Err(e) => {
453                    eprintln!("Error in layer {layer_id}: {e:?}");
454                    return Err(e);
455                }
456            };
457        }
458
459        Ok(current_inputs)
460    }

525    /// Calculate optimization score for DAA coordination
526    fn calculate_optimization_score(&self) -> f32 {
527        if let Ok(tracker) = self.performance_tracker.lock() {
528            let total_operations = tracker.operation_counts.values().sum::<u64>();
529            if total_operations > 0 {
530                let optimization_events = tracker.optimization_events.len() as f32;
531                let efficiency = optimization_events / total_operations as f32;
532                efficiency.min(1.0)
533            } else {
534                1.0
535            }
536        } else {
537            0.0
538        }
539    }

542impl PerformanceTracker {
543    fn new() -> Self {
544        Self {
545            operation_counts: HashMap::new(),
546            execution_times: HashMap::new(),
547            backend_switches: HashMap::new(),
548            optimization_events: Vec::new(),
549        }
550    }
551
552    fn record_operation(&mut self, operation: &str, duration: f64, backend: BackendType) {
553        *self
554            .operation_counts
555            .entry(operation.to_string())
556            .or_insert(0) += 1;
557        self.execution_times
558            .entry(operation.to_string())
559            .or_default()
560            .push(duration);
561        *self.backend_switches.entry(backend).or_insert(0) += 1;
562    }
563
564    fn get_stats(&self) -> TrackerStats {
565        TrackerStats {
566            total_operations: self.operation_counts.values().sum(),
567            average_duration: self
568                .execution_times
569                .values()
570                .flat_map(|times| times.iter())
571                .sum::<f64>()
572                / self
573                    .execution_times
574                    .values()
575                    .map(|times| times.len())
576                    .sum::<usize>() as f64,
577            backend_distribution: self.backend_switches.clone(),
578            optimization_events: self.optimization_events.len(),
579        }
580    }
581}

583/// CPU activation function implementation
584fn apply_activation_cpu<T: Float>(x: T, function: ActivationFunction, steepness: T) -> T {
585    match function {
586        ActivationFunction::Linear => x * steepness,
587        ActivationFunction::Sigmoid => {
588            let exp_val = (-steepness * x).exp();
589            T::one() / (T::one() + exp_val)
590        }
591        ActivationFunction::ReLU => {
592            if x > T::zero() {
593                x
594            } else {
595                T::zero()
596            }
597        }
598        ActivationFunction::ReLULeaky => {
599            let alpha = T::from(0.01).unwrap_or(T::zero());
600            if x > T::zero() {
601                x
602            } else {
603                alpha * x
604            }
605        }
606        ActivationFunction::Tanh => (steepness * x).tanh(),
607        _ => x, // Fallback for other functions
608    }
609}
