78/// CPU-side memory manager for fallback when GPU is not available
79pub struct CpuMemoryManager {
80    buffers: Arc<Mutex<HashMap<BufferHandle, Vec<u8>>>>,
81    buffer_info: Arc<Mutex<HashMap<BufferHandle, BufferInfo>>>,
82    next_id: Arc<Mutex<u64>>,
83    stats: Arc<Mutex<MemoryStats>>,
84}
85
86impl CpuMemoryManager {
87    pub fn new() -> Self {
88        Self {
89            buffers: Arc::new(Mutex::new(HashMap::new())),
90            buffer_info: Arc::new(Mutex::new(HashMap::new())),
91            next_id: Arc::new(Mutex::new(1)),
92            stats: Arc::new(Mutex::new(MemoryStats::default())),
93        }
94    }
95
96    pub fn allocate(&self, size: usize) -> Result<BufferHandle, ComputeError> {
97        let mut buffers = self.buffers.lock().unwrap();
98        let mut buffer_info = self.buffer_info.lock().unwrap();
99        let mut next_id = self.next_id.lock().unwrap();
100        let mut stats = self.stats.lock().unwrap();
101
102        let id = *next_id;
103        *next_id += 1;
104        let handle = BufferHandle(id);
105
106        // Create CPU buffer
107        let buffer = vec![0u8; size];
108        buffers.insert(handle, buffer);
109
110        // Update info
111        let mut info = BufferInfo::new(size);
112        info.in_use = true;
113        buffer_info.insert(handle, info);
114
115        // Update stats
116        stats.total_allocated += size;
117        stats.buffer_count += 1;
118
119        Ok(handle)
120    }
121
122    pub fn deallocate(&self, handle: BufferHandle) -> Result<(), ComputeError> {
123        let mut buffers = self.buffers.lock().unwrap();
124        let mut buffer_info = self.buffer_info.lock().unwrap();
125        let mut stats = self.stats.lock().unwrap();
126
127        if let Some(info) = buffer_info.get(&handle) {
128            stats.total_allocated -= info.size;
129            stats.buffer_count -= 1;
130        }
131
132        buffers.remove(&handle);
133        buffer_info.remove(&handle);
134
135        Ok(())
136    }
137
138    pub fn read(&self, handle: BufferHandle) -> Result<Vec<u8>, ComputeError> {
139        let buffers = self.buffers.lock().unwrap();
140        buffers
141            .get(&handle)
142            .cloned()
143            .ok_or_else(|| ComputeError::General("Buffer not found".to_string()))
144    }
145
146    pub fn write(&self, handle: BufferHandle, data: &[u8]) -> Result<(), ComputeError> {
147        let mut buffers = self.buffers.lock().unwrap();
148        let buffer = buffers
149            .get_mut(&handle)
150            .ok_or_else(|| ComputeError::General("Buffer not found".to_string()))?;
151
152        if buffer.len() != data.len() {
153            return Err(ComputeError::General("Buffer size mismatch".to_string()));
154        }
155
156        buffer.copy_from_slice(data);
157        Ok(())
158    }
159
160    pub fn get_stats(&self) -> MemoryStats {
161        self.stats.lock().unwrap().clone()
162    }
163}
164
165impl Default for CpuMemoryManager {
166    fn default() -> Self {
167        Self::new()
168    }

249            let mut pool = self.pool.lock().unwrap();
250            let mut stats = self.stats.lock().unwrap();

261            let mut stats = self.stats.lock().unwrap();
262
263            if let Some(buffer) = pool.deallocate(handle.id()) {
264                stats.total_allocated = stats.total_allocated.saturating_sub(buffer.size as usize);
265                stats.buffer_count = stats.buffer_count.saturating_sub(1);
266            }
267
268            Ok(())
269        }

272            self.stats.lock().unwrap().clone()
273        }

277/// GPU memory manager that automatically selects between WebGPU and CPU implementations
278pub struct GpuMemoryManager {
279    #[cfg(feature = "gpu")]
280    webgpu_manager: Option<webgpu_memory::WebGPUMemoryManager>,
281    cpu_manager: CpuMemoryManager,
282    buffer_pools: Arc<Mutex<HashMap<BufferSize, VecDeque<BufferHandle>>>>,
283    allocated_buffers: Arc<Mutex<HashMap<BufferHandle, BufferInfo>>>,
284    stats: Arc<Mutex<MemoryManagerStats>>,
285}

287#[derive(Debug, Clone)]
288pub struct MemoryManagerStats {
289    allocations_by_size: HashMap<BufferSize, usize>,
290    deallocations_by_size: HashMap<BufferSize, usize>,
291    pool_hits: HashMap<BufferSize, usize>,
292    pool_misses: HashMap<BufferSize, usize>,
293    total_allocations: usize,
294    total_deallocations: usize,
295    peak_memory_usage: usize,
296    current_pool_sizes: HashMap<BufferSize, usize>,
297}
298
299impl GpuMemoryManager {
300    pub fn new() -> Self {
301        Self {
302            #[cfg(feature = "gpu")]
303            webgpu_manager: None, // Would be initialized with actual GPU device
304            cpu_manager: CpuMemoryManager::new(),
305            buffer_pools: Arc::new(Mutex::new(HashMap::new())),
306            allocated_buffers: Arc::new(Mutex::new(HashMap::new())),
307            stats: Arc::new(Mutex::new(MemoryManagerStats::new())),
308        }
309    }
310
311    pub fn allocate(&self, size: usize) -> Result<BufferHandle, ComputeError> {
312        let size_category = match size {
313            0..=1024 => BufferSize::Small,
314            1025..=1_048_576 => BufferSize::Medium,
315            1_048_577..=104_857_600 => BufferSize::Large,
316            _ => BufferSize::XLarge,
317        };
318
319        // Try to get from pool first
320        {
321            let mut pools = self.buffer_pools.lock().unwrap();
322            let mut allocated = self.allocated_buffers.lock().unwrap();
323            let mut stats = self.stats.lock().unwrap();
324
325            if let Some(pool) = pools.get_mut(&size_category) {
326                while let Some(handle) = pool.pop_front() {
327                    if let Some(info) = allocated.get_mut(&handle) {
328                        // Check if buffer is still valid and right size
329                        if !info.in_use && info.size >= size {
330                            info.in_use = true;
331                            info.last_used = std::time::Instant::now();
332                            info.usage_count += 1;
333
334                            // Update stats
335                            *stats.pool_hits.entry(size_category.clone()).or_insert(0) += 1;
336
337                            return Ok(handle);
338                        }
339                    }
340                }
341            }
342
343            // Update miss stats
344            *stats.pool_misses.entry(size_category.clone()).or_insert(0) += 1;
345        }
346
347        // Allocate new buffer
348        #[cfg(feature = "gpu")]
349        if let Some(ref webgpu) = self.webgpu_manager {
350            let handle = webgpu.allocate_buffer(size)?;
351            self.track_allocation(handle, size);
352            return Ok(handle);
353        }
354
355        // Fallback to CPU
356        let handle = self.cpu_manager.allocate(size)?;
357        self.track_allocation(handle, size);
358        Ok(handle)
359    }
360
361    pub fn deallocate(&self, handle: BufferHandle) -> Result<(), ComputeError> {
362        let mut allocated = self.allocated_buffers.lock().unwrap();
363        let mut pools = self.buffer_pools.lock().unwrap();
364        let mut stats = self.stats.lock().unwrap();
365
366        if let Some(mut info) = allocated.remove(&handle) {
367            info.in_use = false;
368
369            // Add to pool for reuse
370            let pool = pools.entry(info.size_category.clone()).or_default();
371
372            // Limit pool size to prevent unbounded growth
373            const MAX_POOL_SIZE: usize = 100;
374            if pool.len() < MAX_POOL_SIZE {
375                allocated.insert(handle, info.clone());
376                pool.push_back(handle);
377
378                // Update pool size stats
379                *stats
380                    .current_pool_sizes
381                    .entry(info.size_category.clone())
382                    .or_insert(0) = pool.len();
383            } else {
384                // Actually deallocate if pool is full
385                #[cfg(feature = "gpu")]
386                if let Some(ref webgpu) = self.webgpu_manager {
387                    return webgpu.deallocate_buffer(handle);
388                }
389
390                self.cpu_manager.deallocate(handle)?;
391            }
392
393            // Update stats
394            *stats
395                .deallocations_by_size
396                .entry(info.size_category)
397                .or_insert(0) += 1;
398            stats.total_deallocations += 1;
399        }
400
401        Ok(())
402    }
403
404    pub fn read(&self, handle: BufferHandle) -> Result<Vec<u8>, ComputeError> {
405        self.cpu_manager.read(handle)
406    }
407

409        self.cpu_manager.write(handle, data)
410    }

413        #[cfg(feature = "gpu")]
414        if let Some(ref webgpu) = self.webgpu_manager {
415            return webgpu.get_stats();
416        }
417
418        self.cpu_manager.get_stats()
419    }

421    pub fn get_detailed_stats(&self) -> MemoryManagerStats {
422        self.stats.lock().unwrap().clone()
423    }

425    pub fn cleanup_unused_buffers(&self, age_threshold: std::time::Duration) {
426        let mut allocated = self.allocated_buffers.lock().unwrap();
427        let mut pools = self.buffer_pools.lock().unwrap();
428        let now = std::time::Instant::now();
429
430        // Find old unused buffers
431        let mut to_remove = Vec::new();
432        for (handle, info) in allocated.iter() {
433            if !info.in_use && now.duration_since(info.last_used) > age_threshold {
434                to_remove.push(*handle);
435            }
436        }
437
438        // Remove from pools and deallocate
439        for handle in to_remove {
440            // Remove from pool
441            for pool in pools.values_mut() {
442                pool.retain(|&h| h != handle);
443            }
444
445            // Remove from allocated tracking
446            allocated.remove(&handle);
447
448            // Actually deallocate
449            #[cfg(feature = "gpu")]
450            if let Some(ref webgpu) = self.webgpu_manager {
451                let _ = webgpu.deallocate_buffer(handle);
452                continue;
453            }
454
455            let _ = self.cpu_manager.deallocate(handle);
456        }
457    }
458
459    fn track_allocation(&self, handle: BufferHandle, size: usize) {
460        let mut allocated = self.allocated_buffers.lock().unwrap();
461        let mut stats = self.stats.lock().unwrap();
462
463        let mut info = BufferInfo::new(size);
464        info.in_use = true;
465        info.usage_count = 1;
466
467        allocated.insert(handle, info.clone());
468
469        // Update stats
470        *stats
471            .allocations_by_size
472            .entry(info.size_category)
473            .or_insert(0) += 1;
474        stats.total_allocations += 1;
475
476        let current_usage: usize = allocated
477            .values()
478            .filter(|info| info.in_use)
479            .map(|info| info.size)
480            .sum();
481
482        if current_usage > stats.peak_memory_usage {
483            stats.peak_memory_usage = current_usage;
484        }
485    }

593impl EnhancedMemoryStats {
594    /// Get cache hit ratio across all buffer pools
595    pub fn cache_hit_ratio(&self) -> f32 {
596        #[cfg(feature = "gpu")]
597        if let Some(ref stats) = self.pool_stats {
598            return stats.cache_hit_ratio();
599        }
600
601        0.0 // Legacy stats don't track cache hits
602    }
603
604    /// Get total allocated memory in bytes
605    pub fn total_allocated(&self) -> u64 {
606        #[cfg(feature = "gpu")]
607        if let Some(ref stats) = self.pool_stats {
608            return stats.global.total_memory_allocated;
609        }
610
611        if let Some(ref legacy) = self.legacy_stats {
612            return legacy.total_allocated as u64;
613        }
614
615        0
616    }
617
618    /// Get current memory pressure as a ratio (0.0-1.0)
619    pub fn pressure_ratio(&self) -> f32 {
620        #[cfg(feature = "gpu")]
621        if self.current_pressure != MemoryPressure::None {
622            match self.current_pressure {
623                MemoryPressure::None => 0.0,
624                MemoryPressure::Low => 0.2,
625                MemoryPressure::Medium => 0.5,
626                MemoryPressure::High => 0.8,
627                MemoryPressure::Critical => 1.0,
628            }
629        } else if let Some(ref legacy) = self.legacy_stats {
630            if legacy.available > 0 {
631                legacy.total_allocated as f32 / (legacy.total_allocated + legacy.available) as f32
632            } else {
633                0.0
634            }
635        } else {
636            0.0
637        }
638
639        #[cfg(not(feature = "gpu"))]
640        if let Some(ref legacy) = self.legacy_stats {
641            if legacy.available > 0 {
642                legacy.total_allocated as f32 / (legacy.total_allocated + legacy.available) as f32
643            } else {
644                0.0
645            }
646        } else {
647            0.0
648        }
649    }
650
651    /// Get average allocation latency in nanoseconds
652    pub fn avg_allocation_latency_ns(&self) -> u64 {
653        #[cfg(feature = "gpu")]
654        if let Some(ref stats) = self.pool_stats {
655            return stats.global.avg_allocation_latency_ns;
656        }
657
658        1_000_000 // 1ms default for legacy
659    }
660
661    /// Generate performance summary string
662    pub fn performance_summary(&self) -> String {
663        #[cfg(feature = "gpu")]
664        if let Some(ref stats) = self.pool_stats {
665            return stats.performance_summary();
666        }
667
668        format!(
669            "Legacy Memory: {:.1}MB allocated, {:.1}% pressure",
670            self.total_allocated() as f64 / (1024.0 * 1024.0),
671            self.pressure_ratio() * 100.0
672        )
673    }

851    /// Get current memory statistics
852    pub fn get_stats(&self) -> ComputeResult<EnhancedMemoryStats> {
853        // Check cache first
854        #[cfg(feature = "gpu")]
855        {
856            if let Ok(cache) = self.statistics_cache.lock() {
857                if let Some((timestamp, stats)) = cache.as_ref() {
858                    if timestamp.elapsed() < Duration::from_millis(100) {
859                        return Ok(stats.clone());
860                    }
861                }
862            }
863        }
864
865        #[cfg_attr(not(feature = "gpu"), allow(unused_mut))]
866        let mut stats = EnhancedMemoryStats {
867            #[cfg(feature = "gpu")]
868            pool_stats: None,
869            #[cfg(feature = "gpu")]
870            monitoring_stats: None,
871            legacy_stats: None,
872            #[cfg(feature = "gpu")]
873            daa_enabled: self.config.enable_daa,
874            #[cfg(feature = "gpu")]
875            monitoring_enabled: self.config.enable_monitoring,
876            #[cfg(feature = "gpu")]
877            current_pressure: MemoryPressure::None,
878            enhanced_features_available: self.config.enable_advanced_features,
879        };
880
881        #[cfg(feature = "gpu")]
882        {
883            // Get pool statistics if available
884            if let Some(ref pool) = self.advanced_pool {
885                stats.pool_stats = Some(pool.get_statistics());
886            }
887
888            // Get monitoring statistics if available
889            if let Some(ref monitor) = self.pressure_monitor {
890                let report = monitor.generate_report();
891                stats.monitoring_stats = Some(report.monitoring_stats);
892                stats.current_pressure = report.current_pressure;
893            }
894
895            // Cache the results
896            if let Ok(mut cache) = self.statistics_cache.lock() {
897                *cache = Some((Instant::now(), stats.clone()));
898            }
899        }
900
901        Ok(stats)
902    }

949    /// Optimize memory layout for DAA coordination
950    #[cfg(feature = "gpu")]
951    pub fn optimize_for_daa(&self) -> ComputeResult<OptimizationResult> {
952        if !self.config.enable_daa {
953            return Ok(OptimizationResult::default());
954        }
955
956        // Check if enough time has passed since last optimization
957        if let Ok(last_opt) = self.last_optimization.lock() {
958            if last_opt.elapsed() < Duration::from_secs(60) {
959                return Ok(OptimizationResult {
960                    memory_reclaimed: 0,
961                    fragmentation_reduced: 0.0,
962                    performance_improvement: 0.0,
963                    operations_optimized: 0,
964                });
965            }
966        }
967
968        let mut result = OptimizationResult::default();
969
970        // Perform optimization if pool available
971        if let Some(ref pool) = self.advanced_pool {
972            // Get current stats before optimization
973            let stats_before = pool.get_statistics();
974
975            // Perform aggressive cleanup
976            // Advanced buffer pool cleanup - for now just log
977            println!("Performing aggressive cleanup");
978
979            // Get stats after optimization
980            let stats_after = pool.get_statistics();
981
982            // Calculate improvements
983            result.memory_reclaimed = stats_before
984                .global
985                .total_memory_allocated
986                .saturating_sub(stats_after.global.total_memory_allocated);
987
988            result.fragmentation_reduced = if stats_before.global.total_cache_hits > 0 {
989                (stats_after.global.total_cache_hits - stats_before.global.total_cache_hits) as f32
990                    / stats_before.global.total_cache_hits as f32
991            } else {
992                0.0
993            };
994
995            result.operations_optimized = stats_after.global.total_allocations;
996
997            // Update last optimization time
998            if let Ok(mut last_opt) = self.last_optimization.lock() {
999                *last_opt = Instant::now();
1000            }
1001        }
1002
1003        Ok(result)
1004    }
