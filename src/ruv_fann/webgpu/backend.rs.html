13        Self: Sized;

16        Self: Sized;
17    fn capabilities(&self) -> BackendCapabilities;
18    fn backend_type(&self) -> BackendType;

20    /// Core neural network operations

22        &self,
23        matrix: &[T],
24        vector: &[T],
25        rows: usize,
26        cols: usize,
27    ) -> Result<Vec<T>, ComputeError>;
28
29    fn batch_matrix_vector_multiply(
30        &self,
31        matrix: &[T],
32        vectors: &[Vec<T>],
33        rows: usize,
34        cols: usize,
35    ) -> Result<Vec<Vec<T>>, ComputeError>;
36
37    fn apply_activation_function(
38        &self,
39        inputs: &[T],
40        function: ActivationFunction,
41        steepness: T,
42    ) -> Result<Vec<T>, ComputeError>;
43
44    fn vector_operations(&self) -> &dyn VectorOps<T>;
45    fn memory_manager(&self) -> &dyn MemoryManager<T>;

136impl<T: Float + std::fmt::Debug + Send + Sync + 'static> BackendSelector<T> {
137    pub fn new() -> Self {
138        let mut backends = Vec::new();
139
140        // Try to initialize backends in order of preference
141        #[cfg(feature = "gpu")]
142        {
143            // WebGPU backend would be initialized here
144            // Note: This requires async initialization, so we'll handle this differently
145        }
146
147        // Add SIMD backend if available
148        if SimdBackend::<T>::is_available() {
149            if let Ok(simd) = SimdBackend::initialize() {
150                backends.push(Box::new(simd) as Box<dyn ComputeBackend<T>>);
151            }
152        }
153
154        // CPU backend always available as final fallback
155        if let Ok(cpu) = CpuBackend::initialize() {
156            backends.push(Box::new(cpu) as Box<dyn ComputeBackend<T>>);
157        }
158
159        Self {
160            backends,
161            performance_cache: HashMap::new(),
162            fallback_chain: vec![BackendType::WebGPU, BackendType::Simd, BackendType::Cpu],
163        }
164    }
165
166    /// Get available backend types
167    pub fn get_available_backends(&self) -> Vec<BackendType> {
168        self.backends.iter().map(|b| b.backend_type()).collect()
169    }
170
171    /// Get current active backend type
172    pub fn get_current_backend(&self) -> BackendType {
173        // Return the first available backend (highest priority)
174        if !self.backends.is_empty() {
175            self.backends[0].backend_type()
176        } else {
177            BackendType::Cpu // Fallback
178        }
179    }
180
181    /// Set the active backend
182    pub fn set_backend(&mut self, backend_type: BackendType) {
183        // Find the backend and move it to the front if available
184        if let Some(pos) = self
185            .backends
186            .iter()
187            .position(|b| b.backend_type() == backend_type)
188        {
189            // Move the selected backend to the front
190            let backend = self.backends.remove(pos);
191            self.backends.insert(0, backend);
192        }
193        // If backend not available, it will gracefully fall back to available ones
194    }
195
196    /// Select optimal backend for given problem size
197    pub fn select_optimal_backend(&mut self, rows: usize, cols: usize) -> BackendType {
198        let matrix_size = if rows > 1000 || cols > 1000 {
199            MatrixSize::Large
200        } else if rows > 100 || cols > 100 {
201            MatrixSize::Medium
202        } else {
203            MatrixSize::Small
204        };
205
206        let profile = ComputeProfile {
207            matrix_size,
208            batch_size: 1,
209            operation_type: OperationType::Inference,
210        };
211
212        // Use existing selection logic
213        if let Some(backend) = self.select_backend(&profile) {
214            backend.backend_type()
215        } else {
216            // Fallback to first available backend
217            self.get_current_backend()
218        }
219    }
220
221    pub fn select_backend(&self, profile: &ComputeProfile) -> Option<&dyn ComputeBackend<T>> {
222        // Check performance cache first
223        if let Some(backend_type) = self.performance_cache.get(profile) {
224            if let Some(backend) = self.find_backend(*backend_type) {
225                return Some(backend);
226            }
227        }
228
229        // Default selection logic
230        match profile.matrix_size {
231            MatrixSize::Large => self
232                .find_backend(BackendType::WebGPU)
233                .or_else(|| self.find_backend(BackendType::Simd))
234                .or_else(|| self.find_backend(BackendType::Cpu)),
235            MatrixSize::Medium => {
236                if profile.batch_size > 10 {
237                    self.find_backend(BackendType::WebGPU)
238                        .or_else(|| self.find_backend(BackendType::Simd))
239                        .or_else(|| self.find_backend(BackendType::Cpu))
240                } else {
241                    self.find_backend(BackendType::Simd)
242                        .or_else(|| self.find_backend(BackendType::Cpu))
243                }
244            }
245            MatrixSize::Small => self
246                .find_backend(BackendType::Simd)
247                .or_else(|| self.find_backend(BackendType::Cpu)),
248        }
249    }
250
251    fn find_backend(&self, backend_type: BackendType) -> Option<&dyn ComputeBackend<T>> {
252        self.backends
253            .iter()
254            .find(|b| b.backend_type() == backend_type)
255            .map(|b| b.as_ref())
256    }
257
258    pub fn capabilities(&self) -> Vec<BackendCapabilities> {
259        self.backends.iter().map(|b| b.capabilities()).collect()
260    }
261}

349            }
350            result.push(sum);
351        }

353        Ok(result)
354    }

394impl<T: Float + std::fmt::Debug + Send + Sync + 'static> SimdBackend<T> {
395    fn apply_activation_cpu(&self, x: T, function: ActivationFunction, steepness: T) -> T {
396        match function {
397            ActivationFunction::Sigmoid => {
398                let exp_val = (-steepness * x).exp();
399                T::one() / (T::one() + exp_val)
400            }
401            ActivationFunction::ReLU => {
402                if x > T::zero() {
403                    x
404                } else {
405                    T::zero()
406                }
407            }
408            ActivationFunction::ReLULeaky => {
409                let alpha = T::from(0.01).unwrap_or(T::zero());
410                if x > T::zero() {
411                    x
412                } else {
413                    alpha * x
414                }
415            }
416            ActivationFunction::Tanh => {
417                let exp_2x = (steepness * x + steepness * x).exp();
418                let exp_neg_2x = (-steepness * x - steepness * x).exp();
419                (exp_2x - exp_neg_2x) / (exp_2x + exp_neg_2x)
420            }
421            ActivationFunction::Linear => x * steepness,
422            _ => x, // Fallback for other functions
423        }
424    }
425}

428    fn dot_product(&self, a: &[T], b: &[T]) -> Result<T, ComputeError> {
429        if a.len() != b.len() {
430            return Err(ComputeError::InvalidDimensions(

432            ));
433        }
434
435        let mut sum = T::zero();
436        for (x, y) in a.iter().zip(b.iter()) {
437            sum = sum + *x * *y;
438        }
439        Ok(sum)
440    }

448
449        Ok(a.iter().zip(b.iter()).map(|(x, y)| *x + *y).collect())
450    }
