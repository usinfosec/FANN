1//! GPU device management with advanced capabilities detection
2//!
3//! This module provides comprehensive GPU device initialization, capability detection,
4//! and management for the WebGPU backend. It includes staging optimizations for
5//! production-ready GPU acceleration.
6
7use crate::webgpu::error::{ComputeError, ComputeResult};
8
9/// GPU device wrapper with advanced capabilities
10#[derive(Debug)]
11pub struct GpuDevice {
12    /// WebGPU device handle
13    pub device: ::wgpu::Device,
14    /// WebGPU queue for command submission
15    pub queue: ::wgpu::Queue,
16    /// Device adapter information
17    adapter_info: ::wgpu::AdapterInfo,
18    /// Device limits and capabilities
19    limits: ::wgpu::Limits,
20}
21
22/// Device type classification for optimization
23#[derive(Debug, Clone, Copy, PartialEq)]
24pub enum DeviceType {
25    DiscreteGpu,
26    IntegratedGpu,
27    VirtualGpu,
28    Cpu,
29    Unknown,
30}
31
32/// Detailed device information for optimization
33#[derive(Debug, Clone)]
34pub struct DeviceInfo {
35    pub device_type: DeviceType,
36    pub name: String,
37    pub vendor: String,
38    pub backend: String,
39    pub limits: ::wgpu::Limits,
40    pub features: ::wgpu::Features,
41}

43impl GpuDevice {
44    /// Initialize GPU device with advanced capability detection
45    pub async fn new() -> ComputeResult<Self> {
46        // Check if running in CI environment
47        if std::env::var("RUV_FANN_CI_TESTING").is_ok() {
48            // Return a mock device for CI testing
49            return Err(ComputeError::GpuUnavailable);
50        }
51
52        // Create WebGPU instance
53        let instance = ::wgpu::Instance::new(::wgpu::InstanceDescriptor {
54            backends: ::wgpu::Backends::all(),
55            flags: ::wgpu::InstanceFlags::default(),
56            dx12_shader_compiler: ::wgpu::Dx12Compiler::default(),
57            gles_minor_version: ::wgpu::Gles3MinorVersion::Automatic,
58        });
59
60        // Request adapter with high performance preference
61        let adapter = instance
62            .request_adapter(&::wgpu::RequestAdapterOptions {
63                power_preference: ::wgpu::PowerPreference::HighPerformance,
64                compatible_surface: None,
65                force_fallback_adapter: false,
66            })
67            .await
68            .ok_or_else(|| {
69                ComputeError::InitializationError(
70                    "Failed to find suitable WebGPU adapter".to_string(),
71                )
72            })?;
73
74        // Get adapter info for optimization decisions
75        let adapter_info = adapter.get_info();
76
77        // Request device with required features and limits
78        let required_features = ::wgpu::Features::empty();
79        let required_limits = ::wgpu::Limits {
80            max_compute_workgroup_storage_size: 32768,
81            max_compute_workgroups_per_dimension: 65535,
82            max_compute_workgroup_size_x: 1024,
83            max_compute_workgroup_size_y: 1024,
84            max_compute_workgroup_size_z: 64,
85            ..::wgpu::Limits::downlevel_webgl2_defaults()
86        };
87
88        let (device, queue) = adapter
89            .request_device(
90                &::wgpu::DeviceDescriptor {
91                    label: Some("ruv-FANN GPU Device"),
92                    required_features,
93                    required_limits: required_limits.clone(),
94                },
95                None,
96            )
97            .await
98            .map_err(|e| {
99                ComputeError::InitializationError(format!("Failed to create WebGPU device: {}", e))
100            })?;
101
102        Ok(Self {
103            device,
104            queue,
105            adapter_info,
106            limits: required_limits,
107        })
108    }

110    /// Get comprehensive device information
111    pub fn get_info(&self) -> DeviceInfo {
112        let device_type = match self.adapter_info.device_type {
113            ::wgpu::DeviceType::DiscreteGpu => DeviceType::DiscreteGpu,
114            ::wgpu::DeviceType::IntegratedGpu => DeviceType::IntegratedGpu,
115            ::wgpu::DeviceType::VirtualGpu => DeviceType::VirtualGpu,
116            ::wgpu::DeviceType::Cpu => DeviceType::Cpu,
117            ::wgpu::DeviceType::Other => DeviceType::Unknown,
118        };
119
120        DeviceInfo {
121            device_type,
122            name: self.adapter_info.name.clone(),
123            vendor: format!("{:?}", self.adapter_info.vendor),
124            backend: format!("{:?}", self.adapter_info.backend),
125            limits: self.limits.clone(),
126            features: self.device.features(),
127        }
128    }

131    pub fn create_compute_shader(
132        &self,
133        source: &str,

135    ) -> ComputeResult<::wgpu::ShaderModule> {
136        let shader_descriptor = ::wgpu::ShaderModuleDescriptor {

140

186    /// Estimate optimal workgroup size for given problem size
187    pub fn estimate_optimal_workgroup_size(&self, problem_size: usize) -> u32 {
188        let max_size = self.limits.max_compute_workgroup_size_x.min(1024);
189
190        // Common optimal sizes based on GPU architectures
191        let candidates = [32, 64, 128, 256, 512, 1024];
192
193        candidates
194            .iter()
195            .filter(|&&size| size <= max_size)
196            .min_by_key(|&&size| {
197                // Prefer sizes that minimize padding and align with problem size
198                // Using (a + b - 1) / b for ceiling division instead of unstable div_ceil
199                ((problem_size + size as usize - 1) / size as usize) * size as usize - problem_size
200            })
201            .copied()
202            .unwrap_or(64) // Safe fallback
203    }

205    /// Check if device is suitable for high-performance computing
206    pub fn is_high_performance(&self) -> bool {
207        matches!(self.get_info().device_type, DeviceType::DiscreteGpu)
208            && self.limits.max_compute_workgroup_size_x >= 256
209            && self.limits.max_storage_buffer_binding_size >= 128 * 1024 * 1024 // 128MB
210    }

212    /// Get estimated memory bandwidth in GB/s
213    pub fn estimated_memory_bandwidth(&self) -> f32 {
214        match self.get_info().device_type {
215            DeviceType::DiscreteGpu => 500.0,  // Modern discrete GPU
216            DeviceType::IntegratedGpu => 50.0, // Integrated GPU
217            DeviceType::VirtualGpu => 25.0,    // Virtual/cloud GPU
218            DeviceType::Cpu => 25.0,           // CPU memory
219            DeviceType::Unknown => 10.0,       // Conservative fallback
220        }
221    }

223    /// Get estimated compute throughput in GFLOPS
224    pub fn estimated_compute_throughput(&self) -> f32 {
225        let base_throughput = match self.get_info().device_type {
226            DeviceType::DiscreteGpu => 1000.0,
227            DeviceType::IntegratedGpu => 100.0,
228            DeviceType::VirtualGpu => 200.0,
229            DeviceType::Cpu => 50.0,
230            DeviceType::Unknown => 10.0,
231        };
232
233        // Scale by compute units (estimated from workgroup size)
234        let compute_units = self.limits.max_compute_workgroup_size_x as f32 / 32.0;
235        base_throughput * compute_units.min(32.0) // Cap at 32x scaling
236    }

335    /// Validate device capabilities for neural network operations
336    pub fn validate_neural_network_support(&self) -> ComputeResult<()> {
337        let info = self.get_info();
338
339        // Check minimum workgroup size
340        if info.limits.max_compute_workgroup_size_x < 32 {
341            return Err(ComputeError::InitializationError(
342                "Device workgroup size too small for neural network operations".to_string(),
343            ));
344        }
345
346        // Check minimum storage buffer size
347        if info.limits.max_storage_buffer_binding_size < 16 * 1024 * 1024 {
348            // 16MB
349            return Err(ComputeError::InitializationError(
350                "Device storage buffer size too small for neural network operations".to_string(),
351            ));
352        }
353
354        // Check compute invocations
355        if info.limits.max_compute_invocations_per_workgroup < 256 {
356            return Err(ComputeError::InitializationError(
357                "Device compute invocations per workgroup too small".to_string(),
358            ));
359        }
360
361        Ok(())
362    }

377    #[tokio::test]
378    async fn test_device_creation() {
379        // Skip test in CI environment
380        if is_ci_environment() {
381            println!("Skipping WebGPU device test in CI environment");
382            return;
383        }
384
385        // This test will only pass if WebGPU is available
386        match GpuDevice::new().await {
387            Ok(device) => {
388                let info = device.get_info();
389                println!("Device: {} ({:?})", info.name, info.device_type);
390
391                // Check device capabilities with soft assertions
392                // Using unwrap_or for non-fatal error handling
393                assert!(device.max_buffer_size() > 0);
394
395                let storage_buffer_size = device.max_storage_buffer_binding_size();
396                if storage_buffer_size == 0 {
397                    println!("Warning: Device reports zero max_storage_buffer_binding_size!");
398                } else {
399                    println!("Max storage buffer binding size: {}", storage_buffer_size);
400                }
401
402                // Test validation
403                let validation_result = device.validate_neural_network_support();
404                if validation_result.is_ok() {
405                    println!("Device supports neural network operations");
406                } else {
407                    println!("Device limitations: {:?}", validation_result);
408                }
409            }
410            Err(e) => {
411                println!("WebGPU not available: {}", e);
412                // Skip test if WebGPU is not available
