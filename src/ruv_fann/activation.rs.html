4/// Activation functions available for neurons
5///
6/// These functions are based on the FANN library's activation functions
7/// and include both common neural network activation functions and
8/// some specialized variants.
9#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
10#[cfg_attr(feature = "serde", derive(Serialize, Deserialize))]
11#[derive(Default)]
12pub enum ActivationFunction {
13    /// Linear activation function: f(x) = x * steepness
14    Linear,
15
16    /// Threshold activation function: f(x) = 0 if x < 0, 1 if x >= 0
17    /// Note: Cannot be used during training due to zero derivative
18    Threshold,
19
20    /// Symmetric threshold: f(x) = -1 if x < 0, 1 if x >= 0
21    /// Note: Cannot be used during training due to zero derivative
22    ThresholdSymmetric,
23
24    /// Sigmoid activation function: f(x) = 1 / (1 + exp(-2 * steepness * x))
25    /// Output range: (0, 1)
26    #[default]
27    Sigmoid,
28
29    /// Symmetric sigmoid (tanh): f(x) = tanh(steepness * x)
30    /// Output range: (-1, 1)
31    SigmoidSymmetric,
32
33    /// Hyperbolic tangent: alias for SigmoidSymmetric
34    Tanh,
35
36    /// Gaussian activation: f(x) = exp(-x * steepness * x * steepness)
37    /// Output range: (0, 1], peaks at x=0
38    Gaussian,
39
40    /// Symmetric gaussian: f(x) = exp(-x * steepness * x * steepness) * 2 - 1
41    /// Output range: (-1, 1], peaks at x=0
42    GaussianSymmetric,
43
44    /// Elliott activation: f(x) = ((x * steepness) / 2) / (1 + |x * steepness|) + 0.5
45    /// Fast approximation to sigmoid, output range: (0, 1)
46    Elliot,
47
48    /// Symmetric Elliott: f(x) = (x * steepness) / (1 + |x * steepness|)
49    /// Fast approximation to tanh, output range: (-1, 1)
50    ElliotSymmetric,
51
52    /// Bounded linear: f(x) = max(0, min(1, x * steepness))
53    /// Output range: [0, 1]
54    LinearPiece,
55
56    /// Symmetric bounded linear: f(x) = max(-1, min(1, x * steepness))
57    /// Output range: [-1, 1]
58    LinearPieceSymmetric,
59
60    /// Rectified Linear Unit (ReLU): f(x) = max(0, x)
61    /// Output range: [0, ∞)
62    ReLU,
63
64    /// Leaky ReLU: f(x) = x if x > 0, 0.01 * x if x <= 0
65    /// Output range: (-∞, ∞)
66    ReLULeaky,
67
68    /// Sine activation: f(x) = sin(x * steepness) / 2 + 0.5
69    /// Output range: [0, 1]
70    Sin,
71
72    /// Cosine activation: f(x) = cos(x * steepness) / 2 + 0.5
73    /// Output range: [0, 1]
74    Cos,
75
76    /// Symmetric sine: f(x) = sin(x * steepness)
77    /// Output range: [-1, 1]
78    SinSymmetric,
79
80    /// Symmetric cosine: f(x) = cos(x * steepness)
81    /// Output range: [-1, 1]
82    CosSymmetric,
83}
84
85impl ActivationFunction {
86    /// Returns the string name of the activation function
87    pub fn name(&self) -> &'static str {
88        match self {
89            ActivationFunction::Linear => "Linear",
90            ActivationFunction::Threshold => "Threshold",
91            ActivationFunction::ThresholdSymmetric => "ThresholdSymmetric",
92            ActivationFunction::Sigmoid => "Sigmoid",
93            ActivationFunction::SigmoidSymmetric => "SigmoidSymmetric",
94            ActivationFunction::Tanh => "Tanh",
95            ActivationFunction::Gaussian => "Gaussian",
96            ActivationFunction::GaussianSymmetric => "GaussianSymmetric",
97            ActivationFunction::Elliot => "Elliot",
98            ActivationFunction::ElliotSymmetric => "ElliotSymmetric",
99            ActivationFunction::LinearPiece => "LinearPiece",
100            ActivationFunction::LinearPieceSymmetric => "LinearPieceSymmetric",
101            ActivationFunction::ReLU => "ReLU",
102            ActivationFunction::ReLULeaky => "ReLULeaky",
103            ActivationFunction::Sin => "Sin",
104            ActivationFunction::Cos => "Cos",
105            ActivationFunction::SinSymmetric => "SinSymmetric",
106            ActivationFunction::CosSymmetric => "CosSymmetric",
107        }
108    }
109
110    /// Returns whether this activation function can be used during training
111    /// (i.e., has a computable derivative)
112    pub fn is_trainable(&self) -> bool {
113        !matches!(
114            self,
115            ActivationFunction::Threshold | ActivationFunction::ThresholdSymmetric
116        )
117    }
118
119    /// Returns the output range of the activation function
120    pub fn output_range(&self) -> (&'static str, &'static str) {
121        match self {
122            ActivationFunction::Linear => ("-inf", "inf"),
123            ActivationFunction::Threshold => ("0", "1"),
124            ActivationFunction::ThresholdSymmetric => ("-1", "1"),
125            ActivationFunction::Sigmoid => ("0", "1"),
126            ActivationFunction::SigmoidSymmetric | ActivationFunction::Tanh => ("-1", "1"),
127            ActivationFunction::Gaussian => ("0", "1"),
128            ActivationFunction::GaussianSymmetric => ("-1", "1"),
129            ActivationFunction::Elliot => ("0", "1"),
130            ActivationFunction::ElliotSymmetric => ("-1", "1"),
131            ActivationFunction::LinearPiece => ("0", "1"),
132            ActivationFunction::LinearPieceSymmetric => ("-1", "1"),
133            ActivationFunction::ReLU => ("0", "inf"),
134            ActivationFunction::ReLULeaky => ("-inf", "inf"),
135            ActivationFunction::Sin | ActivationFunction::Cos => ("0", "1"),
136            ActivationFunction::SinSymmetric | ActivationFunction::CosSymmetric => ("-1", "1"),
137        }
138    }
139}

141#[cfg(test)]
142mod tests {
143    use super::*;
144
145    #[test]
146    fn test_activation_function_names() {
147        assert_eq!(ActivationFunction::Sigmoid.name(), "Sigmoid");
148        assert_eq!(ActivationFunction::ReLU.name(), "ReLU");
149        assert_eq!(ActivationFunction::Tanh.name(), "Tanh");
150    }
151
152    #[test]
153    fn test_trainable() {
154        assert!(ActivationFunction::Sigmoid.is_trainable());
155        assert!(ActivationFunction::ReLU.is_trainable());
156        assert!(!ActivationFunction::Threshold.is_trainable());
157        assert!(!ActivationFunction::ThresholdSymmetric.is_trainable());
158    }
159
160    #[test]
161    fn test_output_ranges() {
162        assert_eq!(ActivationFunction::Sigmoid.output_range(), ("0", "1"));
163        assert_eq!(ActivationFunction::Tanh.output_range(), ("-1", "1"));
164        assert_eq!(ActivationFunction::ReLU.output_range(), ("0", "inf"));
165        assert_eq!(ActivationFunction::Linear.output_range(), ("-inf", "inf"));
166    }
