19    /// Read a neural network from a FANN format file
20    pub fn read_network<T: Float + std::str::FromStr, R: std::io::Read>(
21        &self,
22        reader: &mut R,
23    ) -> IoResult<Network<T>>
24    where
25        T::Err: std::fmt::Debug,
26    {
27        let mut buf_reader = BufReader::new(reader);
28        let mut line = String::new();
29
30        // Read version line
31        buf_reader.read_line(&mut line)?;
32        if !line.starts_with("FANN_FLO") && !line.starts_with("FANN_FIX") {
33            return Err(IoError::InvalidFileFormat(
34                "Missing FANN version header".to_string(),
35            ));
36        }
37
38        let mut num_layers = 0;
39        let mut connection_rate = T::one();
40        let mut layer_sizes = Vec::new();
41        let mut weights = Vec::new();
42
43        // Parse network parameters
44        loop {
45            line.clear();
46            let bytes_read = buf_reader.read_line(&mut line)?;
47            if bytes_read == 0 {
48                break; // EOF
49            }
50
51            let line = line.trim();
52            if line.is_empty() {
53                continue;
54            }
55
56            if let Some((key, value)) = line.split_once('=') {
57                match key {
58                    "num_layers" => {
59                        num_layers = value.parse().map_err(|e| {
60                            IoError::ParseError(format!("Invalid num_layers: {e:?}"))
61                        })?;
62                    }
63                    "connection_rate" => {
64                        connection_rate = value.parse().map_err(|e| {
65                            IoError::ParseError(format!("Invalid connection_rate: {e:?}"))
66                        })?;
67                    }
68                    "layer_sizes" => {
69                        layer_sizes = value
70                            .split_whitespace()
71                            .map(|s| s.parse())
72                            .collect::<Result<Vec<_>, _>>()
73                            .map_err(|e| {
74                                IoError::ParseError(format!("Invalid layer_sizes: {e:?}"))
75                            })?;
76                    }
77                    "weights" => {
78                        weights = value
79                            .split_whitespace()
80                            .map(|s| s.parse())
81                            .collect::<Result<Vec<_>, _>>()
82                            .map_err(|e| IoError::ParseError(format!("Invalid weights: {e:?}")))?;
83                    }
84                    _ => {
85                        // Skip unknown parameters for now
86                    }
87                }
88            }
89        }
90
91        // Validate network parameters
92        if num_layers == 0 {
93            return Err(IoError::InvalidNetwork(
94                "num_layers must be > 0".to_string(),
95            ));
96        }
97
98        if layer_sizes.is_empty() {
99            return Err(IoError::InvalidNetwork(
100                "layer_sizes must not be empty".to_string(),
101            ));
102        }
103
104        if layer_sizes.len() != num_layers {
105            return Err(IoError::InvalidNetwork(
106                "layer_sizes length must match num_layers".to_string(),
107            ));
108        }
109
110        // Build network using NetworkBuilder
111        let mut builder = NetworkBuilder::<T>::new();
112
113        for (i, &size) in layer_sizes.iter().enumerate() {
114            if i == 0 {
115                builder = builder.input_layer(size);
116            } else if i == layer_sizes.len() - 1 {
117                builder = builder.output_layer(size);
118            } else {
119                builder = builder.hidden_layer(size);
120            }
121        }
122
123        let mut network = builder.connection_rate(connection_rate).build();
124
125        // Set weights if provided
126        if !weights.is_empty() {
127            network
128                .set_weights(&weights)
129                .map_err(|e| IoError::InvalidNetwork(format!("Failed to set weights: {e}")))?;
130        }
131
132        Ok(network)
133    }
134}

153    /// Write a neural network to FANN format
154    pub fn write_network<T: Float + std::fmt::Display, W: Write>(
155        &self,
156        network: &Network<T>,
157        writer: &mut W,
158    ) -> IoResult<()> {
159        // Write version header
160        writeln!(writer, "FANN_FLO:2.1")?;
161
162        // Write network parameters
163        writeln!(writer, "num_layers={}", network.num_layers())?;
164        writeln!(writer, "connection_rate={:.6}", network.connection_rate)?;
165        writeln!(writer, "network_type=0")?;
166        writeln!(writer, "learning_momentum=0.000000")?;
167        writeln!(writer, "training_algorithm=2")?;
168        writeln!(writer, "train_error_function=1")?;
169        writeln!(writer, "train_stop_function=0")?;
170        writeln!(writer, "cascade_output_change_fraction=0.010000")?;
171        writeln!(writer, "quickprop_decay=-0.000100")?;
172        writeln!(writer, "quickprop_mu=1.750000")?;
173        writeln!(writer, "rprop_increase_factor=1.200000")?;
174        writeln!(writer, "rprop_decrease_factor=0.500000")?;
175        writeln!(writer, "rprop_delta_min=0.000000")?;
176        writeln!(writer, "rprop_delta_max=50.000000")?;
177        writeln!(writer, "rprop_delta_zero=0.100000")?;
178        writeln!(writer, "cascade_output_stagnation_epochs=12")?;
179        writeln!(writer, "cascade_candidate_change_fraction=0.010000")?;
180        writeln!(writer, "cascade_candidate_stagnation_epochs=12")?;
181        writeln!(writer, "cascade_max_out_epochs=150")?;
182        writeln!(writer, "cascade_min_out_epochs=50")?;
183        writeln!(writer, "cascade_max_cand_epochs=150")?;
184        writeln!(writer, "cascade_min_cand_epochs=50")?;
185        writeln!(writer, "cascade_num_candidate_groups=2")?;
186        writeln!(writer, "bit_fail_limit=0.350000")?;
187        writeln!(writer, "cascade_candidate_limit=1000.000000")?;
188        writeln!(writer, "cascade_weight_multiplier=0.400000")?;
189        writeln!(writer, "cascade_activation_functions_count=10")?;
190        writeln!(
191            writer,
192            "cascade_activation_functions=3 5 7 8 10 11 14 15 16 17 "
193        )?;
194        writeln!(writer, "cascade_activation_steepnesses_count=4")?;
195        writeln!(
196            writer,
197            "cascade_activation_steepnesses=0.250000 0.500000 0.750000 1.000000 "
198        )?;
199
200        // Write layer sizes
201        write!(writer, "layer_sizes=")?;
202        for layer in &network.layers {
203            write!(writer, "{} ", layer.num_regular_neurons())?;
204        }
205        writeln!(writer)?;
206
207        // Write weights
208        let weights = network.get_weights();
209        if !weights.is_empty() {
210            write!(writer, "weights=")?;
211            for weight in weights {
212                write!(writer, "{weight:.6} ")?;
213            }
214            writeln!(writer)?;
215        }
216
217        Ok(())
218    }
219}
