1//! Error types for I/O operations
2
3use std::fmt;
4use std::io;
5
6/// Result type for I/O operations
7pub type IoResult<T> = Result<T, IoError>;
8
9/// Error types for I/O operations
10#[derive(Debug)]
11pub enum IoError {
12    /// I/O error (file not found, permission denied, etc.)
13    Io(io::Error),
14    /// Invalid file format
15    InvalidFileFormat(String),
16    /// Parse error
17    ParseError(String),
18    /// Serialization error
19    SerializationError(String),
20    /// Decompression error
21    CompressionError(String),
22    /// Invalid network structure
23    InvalidNetwork(String),
24    /// Invalid training data
25    InvalidTrainingData(String),
26}
27
28impl fmt::Display for IoError {
29    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
30        match self {
31            IoError::Io(err) => write!(f, "I/O error: {err}"),
32            IoError::InvalidFileFormat(msg) => write!(f, "Invalid file format: {msg}"),
33            IoError::ParseError(msg) => write!(f, "Parse error: {msg}"),
34            IoError::SerializationError(msg) => write!(f, "Serialization error: {msg}"),
35            IoError::CompressionError(msg) => write!(f, "Compression error: {msg}"),
36            IoError::InvalidNetwork(msg) => write!(f, "Invalid network: {msg}"),
37            IoError::InvalidTrainingData(msg) => write!(f, "Invalid training data: {msg}"),
38        }
39    }
40}
41
42impl std::error::Error for IoError {
43    fn source(&self) -> Option<&(dyn std::error::Error + 'static)> {
44        match self {
45            IoError::Io(err) => Some(err),
46            _ => None,
47        }
48    }
49}
50
51impl From<io::Error> for IoError {
52    fn from(err: io::Error) -> Self {
53        IoError::Io(err)
54    }
55}
56
57#[cfg(feature = "serde")]
58impl From<serde_json::Error> for IoError {
59    fn from(err: serde_json::Error) -> Self {
60        IoError::SerializationError(err.to_string())
61    }
62}
63
64#[cfg(feature = "binary")]
65impl From<bincode::Error> for IoError {
66    fn from(err: bincode::Error) -> Self {
67        IoError::SerializationError(format!("Bincode error: {err}"))
68    }
69}
70
71impl From<std::num::ParseFloatError> for IoError {
72    fn from(err: std::num::ParseFloatError) -> Self {
73        IoError::ParseError(format!("Float parse error: {err}"))
74    }
75}
76
77impl From<std::num::ParseIntError> for IoError {
78    fn from(err: std::num::ParseIntError) -> Self {
79        IoError::ParseError(format!("Integer parse error: {err}"))
80    }
